import { parsePlayers } from './players';

/**
 * Finds the row index for a specific player in the master 'Players' sheet.
 * Uses the dynamic parser to ensure it works even if sheet columns move.
 */
export function findPlayerRowIndex(
  rows: string[][],
  target: { identity: string }
): number {
  if (!rows || rows.length < 2) {
    throw new Error('No data found in sheet');
  }

  // Use your existing parser to get players with stable identities
  const parsedPlayers = parsePlayers(rows);

  // Find the player in the parsed array
  const foundIndex = parsedPlayers.findIndex(
    (p) => p.identity.toLowerCase().trim() === target.identity.toLowerCase().trim()
  );

  if (foundIndex === -1) {
    console.error("Lookup Failed for identity:", target.identity);
    throw new Error('Player row index not found');
  }

  // We need to map the index in parsedPlayers back to the physical row in the Google Sheet.
  // 1. findIndex is 0-based.
  // 2. parsePlayers skips the header row.
  // 3. Google Sheets is 1-based.
  // result: index + 2 (header + shift from 0 to 1)
  return foundIndex + 2;
}
